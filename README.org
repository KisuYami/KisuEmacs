** .xinitrc
   #+BEGIN_SRC bash
#!/bin/sh
xset r rate 200 50
(sleep 0.001 && mpd) &
(sleep 0.001 && xmodmap ~/.editorkeys) &
(sleep 0.001 && xrdb .Xresources) &
(sleep 0.001 && unclutter ) &
(sleep 0.001 && eval $(/usr/bin/gnome-keyring-daemon --start --components=pkcs11,secrets,ssh)) &
(sleep 0.001 && export SSH_AUTH_SOCK) &
(sleep 0.001 && xmodmap ~/.editorkeys) &
exec emacs
   #+END_SRC
